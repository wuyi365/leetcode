            // 提交举报
            var submitReport = function(event) {
                // 获取举报游戏编号，原因
                var gindex = $('input[name="gindex"]').val(),
                    reportComment = $('#report-note').val(),
                    rtype = $("input[name='type']").val(),
                    bid   = $("input[name='bid']").val(),
                    reportReason = new Array();
                    
                // 将举报内容放到数组中
                $('input[name="report-reason"]:checked').each(function(){
                    reportReason.push($(this).val());
                });
                
                $.ajax({
                    url :  '/ajax/game/game_report_submit.json',
                    type : 'post',
                    dataType : 'json',
                    data : {'gindex':gindex, 'report_reason':reportReason, 'report_comment':reportComment,'bid':bid,'type':rtype},
                    success : function (data) {
                        var status = data.status,msg = data.msg;
                        alert(msg);
                        if (status > -1 ) {
                            // 举报成功后，蒙版消失
                            $('#report-modal').modal('hide');
                            // 重置
                            $("input[type=reset]").trigger("click");
                        }
                    },
                    error: function (data, status, e) {
                        alert('服务器繁忙！请稍后再操作');
                    }
                });
            }
            
            $(function() {
                // 点击举报举报按钮
                $('#js-submit-report').on('click',submitReport);
            });
